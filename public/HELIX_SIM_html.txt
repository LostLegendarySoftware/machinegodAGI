                });

                    const throughput = Math.floor(Math.random() * 100) + 50;
                    networkThroughput.textContent = `${throughput} GB/s`;
                    
                    // Add to totals
                    totalInput += throughput;
                    const compressedOutput = throughput / (Math.floor(Math.random() * 300) + 300);
                    totalOutput += compressedOutput;
                    
                    // Update displayed values
                    if (totalInput < 1024) {
                        inputSize.textContent = `${totalInput.toFixed(1)} GB`;
                    } else {
                        inputSize.textContent = `${(totalInput / 1024).toFixed(2)} TB`;

                    totalEnergy += energySavedThisInterval;
                    
                    if (totalEnergy < 1000) {
                        energySaved.textContent = `${totalEnergy.toFixed(1)} kWh`;
                    } else {
                        energySaved.textContent = `${(totalEnergy / 1000).toFixed(1)} MWh`;
                    }
                    
                    // Update global energy counter
                    let globalEnergy = parseFloat(energySavedCounter.textContent);
                    if (isNaN(globalEnergy)) globalEnergy = 0;
                    globalEnergy += energySavedThisInterval / 1000; // convert to MWh
                    energySavedCounter.textContent = `${globalEnergy.toFixed(1)} MWh`;
                    
                    // Randomly add log messages
                    if (Math.random() < 0.3) {
                        const messages = [
                            `Processing batch ${Math.floor(Math.random() * 100000)}... Complete!`,

                        appendToConsole(messages[Math.floor(Math.random() * messages.length)]);
                    }
                }, 1000);
            });

                clearInterval(simulationInterval);
                simulationRunning = false;
